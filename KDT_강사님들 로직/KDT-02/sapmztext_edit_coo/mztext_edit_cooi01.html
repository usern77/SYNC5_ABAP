MODULE USER_COMMAND_0100 INPUT.
  GV_MODE = OK_CODE.
  CASE OK_CODE.
    WHEN 'BACK' OR 'CANC'.
      LEAVE TO SCREEN 0.
    WHEN 'EXIT'.
      LEAVE PROGRAM.
    WHEN 'CHG' OR 'CRT'.
      PERFORM CALL_SCREEN_MODAL.
    WHEN 'SCH'.
      PERFORM GET_DATA.
  ENDCASE.
ENDMODULE.
