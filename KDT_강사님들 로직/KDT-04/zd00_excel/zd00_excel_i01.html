MODULE user_command_0100 INPUT.

  DATA : lv_chk.

  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'SAVE'.
      PERFORM pop_up_to_confirm CHANGING lv_chk.
      CHECK lv_chk = '1'.
      PERFORM insert_data.
  ENDCASE.

ENDMODULE.

MODULE exit INPUT.
  CASE ok_code.

    WHEN 'EXIT'.

  ENDCASE.
ENDMODULE.
