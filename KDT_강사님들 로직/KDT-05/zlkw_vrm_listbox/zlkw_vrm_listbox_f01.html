FORM f4_carrid .

  DATA : ls_list TYPE vrm_value,
         lt_list LIKE TABLE OF ls_list.

  CLEAR   : ls_list.
  REFRESH : lt_list.

  SELECT carrid, carrname
    INTO TABLE @DATA(lt_scarr)
    FROM scarr.

  LOOP AT lt_scarr INTO DATA(ls_scarr).

    ls_list-key  = ls_scarr-carrid.
    ls_list-text = ls_scarr-carrname.

    APPEND ls_list TO lt_list.
    CLEAR  ls_list.

  ENDLOOP.

  CALL FUNCTION 'VRM_SET_VALUES'
    EXPORTING
      id     = 'PA_CARR'
      values = lt_list.

ENDFORM.
